i: integer;
soma, media: real;

writeln ('Introduza os valores das 6 temperaturas');

readln (temp[i])

media := soma / 6;
FOR i := 1 TO 6 DO {Valores acima da média}

writeln ('Temperatura acima da média: ', temp[i]);
END;
